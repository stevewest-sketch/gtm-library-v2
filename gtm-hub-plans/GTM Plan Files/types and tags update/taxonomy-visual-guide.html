‚öôÔ∏è Display Name Formatting

Convert slug values to proper display names

// Standard conversion: 'product-training' ‚Üí 'Product Training'
function formatDisplayName(value) {
  const specialCases = {
    'gtm-training': 'GTM Training',
    'gtm-tool': 'GTM Tool',
    'one-pager': 'One-Pager',
    'bva': 'BVA',
    'roi': 'ROI',
    'csm': 'CSM',
    'ae': 'AE',
    'sc': 'SC',
    'ps': 'PS',
  };
  
  if (specialCases[value]) return specialCases[value];
  
  return value
    .split('-')
    .map(word => word.charAt(0).toUpperCase() + word.slice(1))
    .join(' ');
}
